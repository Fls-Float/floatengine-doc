149 EASEDEF float EaseElasticIn(float t, float b, float c, float d) // Ease: Elastic In
150 {
151 if (t == 0.0f) return b;
152 if ((t /= d) == 1.0f) return (b + c);
153
154 float p = d * 0.3f;
155 float a = c;
156 float s = p / 4.0f;
157 float postFix = a * powf(2.0f, 10.0f * (t -= 1.0f));
158
159 return (-(postFix * sinf((t * d - s) * (2.0f * PI) / p)) + b);
160 }
